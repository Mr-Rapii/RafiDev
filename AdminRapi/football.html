// Kita tidak hardcode API Key di sini async function fetchMatch() { // Mengambil key dari kotak input (Nanti kita buat kotak inputnya) const API_KEY = document.getElementById('apiKeyInput').value; const TEAM_ID = '1432'; const container = document.getElementById('match-container'); const loader = document.getElementById('loader'); if(!API_KEY) { alert("Masukkan API Key dulu!"); return; } container.innerHTML = ''; loader.style.display = 'block'; try { const response = await fetch(`https://v3.football.api-sports.io/fixtures?team=${TEAM_ID}&next=3`, { headers: { 'x-apisports-key': API_KEY } }); const data = await response.json(); loader.style.display = 'none'; if (data.response && data.response.length > 0) { data.response.forEach(match => { const date = new Date(match.fixture.date).toLocaleDateString('id-ID'); container.innerHTML += `

${match.teams.home.name} vs ${match.teams.away.name}

📅 ${date}

`; }); } else { container.innerHTML = "

Data tidak ditemukan/Key salah.

"; } } catch (error) { loader.style.display = 'none'; container.innerHTML = "

Gagal memuat data.

"; } }
